CREATE DATABASE ESTOQUE;

USE ESTOQUE;

CREATE TABLE PRODUTO(ID_PRODUTO INT(5) PRIMARY KEY AUTO_INCREMENT, NM_PRODUTO VARCHAR(45) NOT NULL, DS_PRODUTO VARCHAR(45), VL_PRODUTO DOUBLE(9,2) NOT NULL);

CREATE TABLE CLIENTE(ID_CLIENTE INT(5) PRIMARY KEY AUTO_INCREMENT, NM_CLIENTE VARCHAR(45) NOT NULL, EMAIL_CLIENTE VARCHAR(45), TEL_CLIENTE VARCHAR(45) NOT NULL);

CREATE TABLE PEDIDO(ID_CLIENTE INT(5) NOT NULL, ID_PRODUTO INT(5) NOT NULL, FOREIGH KEY(ID_CLIENTE) REFERENCES CLIENTE(ID_CLIENTE),FOREIGH KEY(ID_PRODUTO) REFERENCES PRODUTO(ID_PRODUTO));
